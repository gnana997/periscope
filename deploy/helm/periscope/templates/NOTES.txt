  Secret mode: {{ .Values.secrets.mode }}

{{- if eq .Values.secrets.mode "existing" }}

  ⚠ secrets.mode=existing requires that the Secret already exists:
    kubectl -n {{ .Release.Namespace }} get secret {{ .Values.secrets.existing.name }}
  If it doesn't, the Pod will CrashLoopBackOff until you apply it. Example:
    kubectl -n {{ .Release.Namespace }} create secret generic {{ .Values.secrets.existing.name }} \
      --from-literal=OIDC_CLIENT_SECRET='<your client secret>'
{{- end }}

{{- if .Values.podIdentity.enabled }}

      --namespace {{ .Release.Namespace }} \
      --service-account {{ include "periscope.serviceAccountName" . }} \
      --role-arn arn:aws:iam::<acct-id>:role/periscope-base
{{- end }}
